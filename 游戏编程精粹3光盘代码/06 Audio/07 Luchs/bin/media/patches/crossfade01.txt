

CGAPatch "Crossfade"
{
	CGASineOscillator LFO1
	{
		Frequency = 0.25;
	}

	CGASineOscillator VCO1
	{
		Frequency = 200;
	}

	CGASineOscillator VCO2
	{
		Frequency = 400;
	}


	CGACrossFade2 Fade
	{
	
	}

	Connection = LFO1,Fade, SampleOut, Balance;
	Connection = VCO1,Fade, SampleOut, SampleIn1;
	Connection = VCO2,Fade, SampleOut, SampleIn2;

}
