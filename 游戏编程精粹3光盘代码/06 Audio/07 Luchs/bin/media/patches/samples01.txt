CGASignal  Impulse
{
	File = "m_impulse07.wav";
}

CGAPatch "Motorcycle Sequencer"
{

	CGATriangleEnvelope Env1
	{
		Duration = 3;
	}

	CGASequencer16 SEQ1
	{
		Frequency = 3.0;
		Steps = Xooo0oooXooo0ooo;
		FMAttenuation = 0.6;
	}


	CGAWavetableOscillator WAV1
	{
		WaveTable = Impulse;
		Repeat = 1;
	}

	Connection = Env1, SEQ1, SampleOut, FM;
	Connection = SEQ1, WAV1, SampleOut, TRIG;

}
